ข้อ1.เป็นการสร้าง function ที่รับ rest parameter (...num) เพื่อรับ argument จำนวนเท่าไหร่ก็ได้ และเก็บไว้ในรูป array ใช้ for loop  
function sum(...num){
    let result = 0
    for(let i = 0; i < arguments.length; i++) //ถ้า i น้อยกว่าจำนวนของ arguments ทั้งหมดก็จะให้ + num[i] เข้าไปใน result ไปเรื่อยๆจนครบ {
        result = result + num[i]
    }
    return result
}

ข้อ3.เป็นการสร้าง function ซ้อน function โดยการสร้าง function calculate ที่มี function add(a,b){return a+b} อยู่ด้านใน แล้วสุดท้ายเรียกใช้ function add(x,y) * 2 เป็นการจบการทำงาน
function calculate(x, y) {
    function add(a, b) {  // ประกาศฟังก์ชัน add
        return a + b
    }
    return add(x, y) * 2  // เรียกใช้ฟังก์ชัน add
}
